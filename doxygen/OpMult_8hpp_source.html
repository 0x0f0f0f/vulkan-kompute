119 {
120  SPDLOG_DEBUG("Kompute OpMult init called");
121 
122  if (this->mTensors.size() < 3) {
123  throw std::runtime_error(
124  "Kompute OpMult called with less than 1 tensor");
125  } else if (this->mTensors.size() > 3) {
126  spdlog::warn("Kompute OpMult called with more than 3 this->mTensors");
127  }
128 
129  this->mTensorLHS = this->mTensors[0];
130  this->mTensorRHS = this->mTensors[1];
131  this->mTensorOutput = this->mTensors[2];
132 
133  // The dispatch size is set up based on either explicitly provided template
134  // parameters or by default it would take the shape and size of the tensors
135  if (tX > 0) {
136  // If at least the x value is provided we use mainly the parameters
137  // provided
138  this->mX = tX;
139  this->mY = tY > 0 ? tY : 1;
140  this->mZ = tZ > 0 ? tZ : 1;
141  } else {
142  // TODO: Fully support the full size dispatch using size for the shape
143  this->mX = this->mTensorLHS->size();
144  this->mY = 1;
145  this->mZ = 1;
146  }
147  spdlog::info("Kompute OpMult dispatch size X: {}, Y: {}, Z: {}",
148  this->mX,
149  this->mY,
150  this->mZ);
151 
152  // TODO: Explore adding a validate function
153  if (!(this->mTensorLHS->isInit() && this->mTensorRHS->isInit() &&
154  this->mTensorOutput->isInit())) {
155  throw std::runtime_error(
156  "Kompute OpMult all tensor parameters must be initialised. LHS: " +
157  std::to_string(this->mTensorLHS->isInit()) +
158  " RHS: " + std::to_string(this->mTensorRHS->isInit()) +
159  " Output: " + std::to_string(this->mTensorOutput->isInit()));
160  }
161 
162  // TODO: Explore use-cases where tensors shouldn't be the same size, and how
163  // to deal with those situations
164  if (!(this->mTensorLHS->size() == this->mTensorRHS->size() &&
165  this->mTensorRHS->size() == this->mTensorOutput->size())) {
166  throw std::runtime_error(
167  "Kompute OpMult all tensor parameters must be the same size LHS: " +
168  std::to_string(this->mTensorLHS->size()) +
169  " RHS: " + std::to_string(this->mTensorRHS->size()) +
170  " Output: " + std::to_string(this->mTensorOutput->size()));
171  }
172 
173  this->mTensorOutputStaging = std::make_shared<Tensor>(
174  this->mTensorOutput->data(), Tensor::TensorTypes::eStaging);
175 
176  this->mTensorOutputStaging->init(
177  this->mPhysicalDevice, this->mDevice, this->mCommandBuffer);
178 
179 #if RELEASE
180  std::vector<char> shaderFileData(
181  shader_data::shaders_glsl_opmult_comp_spv,
182  shader_data::shaders_glsl_opmult_comp_spv +
183  kp::shader_data::shaders_glsl_opmult_comp_spv_len);

209 void
210 OpMult<tX, tY, tZ>::record()
211 {
212  SPDLOG_DEBUG("Kompute OpMult record called");
213 
214  // Barrier to ensure the data is finished writing to buffer memory
215  this->mTensorLHS->recordBufferMemoryBarrier(
216  vk::AccessFlagBits::eHostWrite,
217  vk::AccessFlagBits::eShaderRead,
218  vk::PipelineStageFlagBits::eHost,
219  vk::PipelineStageFlagBits::eComputeShader);
220  this->mTensorRHS->recordBufferMemoryBarrier(
221  vk::AccessFlagBits::eHostWrite,
222  vk::AccessFlagBits::eShaderRead,
223  vk::PipelineStageFlagBits::eHost,
224  vk::PipelineStageFlagBits::eComputeShader);
225 
226  this->mAlgorithm->recordDispatch(this->mX, this->mY, this->mZ);
227 
228  // Barrier to ensure the shader code is executed before buffer read
229  this->mTensorLHS->recordBufferMemoryBarrier(
230  vk::AccessFlagBits::eShaderWrite,
231  vk::AccessFlagBits::eTransferRead,
232  vk::PipelineStageFlagBits::eComputeShader,
233  vk::PipelineStageFlagBits::eTransfer);
234  this->mTensorOutput->recordBufferMemoryBarrier(
235  vk::AccessFlagBits::eShaderWrite,
236  vk::AccessFlagBits::eTransferRead,
237  vk::PipelineStageFlagBits::eComputeShader,
238  vk::PipelineStageFlagBits::eTransfer);
239 
240  this->mTensorOutputStaging->recordCopyFrom(this->mTensorOutput);
241 
242  // Buffer to ensure wait until data is copied to staging buffer
243  this->mTensorLHS->recordBufferMemoryBarrier(
244  vk::AccessFlagBits::eTransferWrite,
245  vk::AccessFlagBits::eHostRead,
246  vk::PipelineStageFlagBits::eTransfer,
247  vk::PipelineStageFlagBits::eHost);
248  this->mTensorOutput->recordBufferMemoryBarrier(
249  vk::AccessFlagBits::eTransferWrite,
250  vk::AccessFlagBits::eHostRead,
251  vk::PipelineStageFlagBits::eTransfer,
252  vk::PipelineStageFlagBits::eHost);
